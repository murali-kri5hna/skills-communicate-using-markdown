## Adding code snippet
```python
import time
import os

def animate_text(text, cycles=3):
    for _ in range(cycles):
        for frame in range(10):
            os.system('cls' if os.name == 'nt' else 'clear')
            offset = frame % len(text)
            print(text[offset:] + text[:offset])
            time.sleep(0.1)

animate_text("* Hello, Python is amazing! *")
```
